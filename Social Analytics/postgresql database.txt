CREATE TABLE tweet (
  id serial PRIMARY KEY,
  search_val text NOT NULL,
  created_at TIMESTAMP,
  created_at_date date,
  created_at_hour text,
  created_at_min text,
  tweet_id text,
  text text,
  source text,
  in_reply_to_status_id text,
  in_reply_to_user_id text,
  in_reply_to_screen_name text,
  user_name text,
  user_screen_name text,
  user_location text,
  user_url text,
  user_description text,
  user_verified integer DEFAULT NULL,
  user_followers_count integer DEFAULT NULL,
  user_friends_count integer DEFAULT NULL,
  user_listed_count integer DEFAULT NULL,
  user_favourites_count integer DEFAULT NULL,
  user_created_at TIMESTAMP,
  user_created_at_date date,
  user_id varchar(50) DEFAULT NULL,
  user_profile_image_url_https text,
  coordinates_lat text,
  coordinates_lon text,
  place_country text,
  place_country_code text,
  place_full_name text,
  place_id varchar(50) DEFAULT NULL,
  place_type varchar(30) DEFAULT NULL,
  quoted_status_id varchar(50) DEFAULT NULL,
  quoted_status text,
  retweeted_status text,
  quote_count integer DEFAULT NULL,
  reply_count integer DEFAULT NULL,
  retweet_count integer DEFAULT NULL,
  favorite_count integer DEFAULT NULL,
  retweeted integer DEFAULT NULL,
  entities text,
  lang varchar(30) DEFAULT NULL,
  feeds_link text,
  feeds_video text,
  feeds_image text,
  clean_text	text,
  text_sentiment_polarity	DOUBLE PRECISION,
  text_sentiment_subjectivity DOUBLE PRECISION,
  created_date TIMESTAMP DEFAULT NOW()
);
