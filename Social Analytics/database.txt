CREATE DATABASE IF NOT EXISTS social_analytic; 
CREATE TABLE `tweet` (
  `id` int NOT NULL AUTO_INCREMENT UNIQUE KEY,
  `search_val` text NOT NULL,
  `created_at` datetime,
  `created_at_date` date,
  `created_at_hour` text,
  `created_at_min` text,
  `tweet_id` text,
  `text` text,
  `source` text,
  `in_reply_to_status_id` text,
  `in_reply_to_user_id` text,
  `in_reply_to_screen_name` text,
  `user_name` text,
  `user_screen_name` text,
  `user_location` text,
  `user_url` text,
  `user_description` text,
  `user_verified` tinyint DEFAULT NULL,
  `user_followers_count` int DEFAULT NULL,
  `user_friends_count` int DEFAULT NULL,
  `user_listed_count` int DEFAULT NULL,
  `user_favourites_count` int DEFAULT NULL,
  `user_created_at` datetime,
  `user_created_at_date` date,
  `user_id` varchar(50) DEFAULT NULL,
  `user_profile_image_url_https` text,
  `coordinates_lat` text,
  `coordinates_lon` text,
  `place_country` text,
  `place_country_code` text,
  `place_full_name` text,
  `place_id` varchar(50) DEFAULT NULL,
  `place_type` varchar(30) DEFAULT NULL,
  `quoted_status_id` varchar(50) DEFAULT NULL,
  `quoted_status` text,
  `retweeted_status` text,
  `quote_count` int DEFAULT NULL,
  `reply_count` int DEFAULT NULL,
  `retweet_count` int DEFAULT NULL,
  `favorite_count` int DEFAULT NULL,
  `retweeted` tinyint DEFAULT NULL,
  `entities` text,
  `lang` varchar(30) DEFAULT NULL,
  `feeds_link` text,
  `feeds_video` text,
  `feeds_image` text,
  `clean_text`	text,
  `text_sentiment_polarity`	double,
  `text_sentiment_subjectivity` double,
  `created_date` datetime DEFAULT CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=latin1;
